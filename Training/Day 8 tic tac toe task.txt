<script>
  let board = ["", "", "", "", "", "", "", "", ""];
  let player = "X";

  function move(pos) {
    if (board[pos] === "") {
      board[pos] = player;
      document.getElementsByTagName("button")[pos].innerText = player;
      if (checkWin()) {
        document.getElementById("msg").innerText = player + " wins!";
      } else {
        player = (player === "X") ? "O" : "X";
      }
    }
  }

  function checkWin() {
    let w = [
      [0,1,2],[3,4,5],[6,7,8],
      [0,3,6],[1,4,7],[2,5,8],
      [0,4,8],[2,4,6]
    ];
    return w.some(c =>
      board[c[0]] === player &&
      board[c[1]] === player &&
      board[c[2]] === player
    );
  }
</script>
